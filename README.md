#### 安装
```bash
npm i yzh-zero-ui -S
```

#### 使用(全局注册所有组件)

```javascript
import {createApp} from 'vue';
import zui from 'yzh-zero-ui';
import 'yzh-zero-ui/dist/style.css';
createApp(App).use(zui);
```

#### 使用(按需引入)

```
import 'yzh-zero-ui/dist/style.css';
import {zeroSlider} from 'yzh-zero-ui';
```
